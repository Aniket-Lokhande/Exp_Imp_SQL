select commodity, trd_dft_pct_change from output_schema.cmd_trd_dft where pct_change_yr_compr = 1 and year = 201920 order by trd_dft_pct_change desc limit 1;

WITH valid_growth AS (
    SELECT
        commodity,
        hscode,
        year,
        exp_pct_change
    FROM output_schema.cmd_trd_dft
    where pct_change_yr_compr = 1
)
SELECT commodity
FROM valid_growth
GROUP BY hscode,commodity
HAVING COUNT(*) = COUNT(CASE WHEN exp_pct_change > 0 THEN 1 END);
